class GuestRegistration {
    static register() {
        try {
            // 1. 간단한 게스트 ID 생성
            const guestId = this.generateGuestId();
            
            // 2. 세션 데이터 생성
            const sessionData = {
                userId: guestId,
                registrationTime: Date.now(),
                userType: 'guest',
                status: 'active'
            };
            
            // 3. LocalStorage 저장 (즉시)
            localStorage.setItem('user_session', JSON.stringify(sessionData));
            
            // 4. 중앙 아키텍처 등록
            CentralDataManager.setData('currentUser', sessionData);
            
            // 5. 성공 로그
            console.log('Guest registration successful:', guestId);
            
            return { success: true, userId: guestId };
            
        } catch (error) {
            console.error('Guest registration failed:', error);
            return { success: false, error: error.message };
        }
    }
    
    static generateGuestId() {
        return `guest_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
    }
}

#### **간결한 구현 예시**
```javascript
// 중앙 데이터 관리자 (ACIU 교훈 적용)
class CentralDataManager {
    constructor() {
        this.data = new Map();
        this.subscribers = new Map();
        this.eventQueue = [];
        this.isProcessing = false;
    }
    
    // 모든 데이터 변경은 이 메서드를 통해서만
    setData(key, value) {
        try {
            // 1. 데이터 저장
            this.data.set(key, value);
            
            // 2. 구독자 알림
            this.notifySubscribers(key, value);
            
            // 3. UI 업데이트 트리거
            this.queueUIUpdate(key);
            
            // 4. 로그 기록
            console.log(`Data updated: ${key}`, value);
            
        } catch (error) {
            console.error('Data update failed:', error);
            throw error;
        }
    }
    
    // 구독자 등록
    subscribe(key, callback) {
        if (!this.subscribers.has(key)) {
            this.subscribers.set(key, []);
        }
        this.subscribers.get(key).push(callback);
    }
    
    // 구독자 알림
    notifySubscribers(key, value) {
        const callbacks = this.subscribers.get(key) || [];
        callbacks.forEach(callback => {
            try {
                callback(value);
            } catch (error) {
                console.error('Subscriber callback failed:', error);
            }
        });
    }
    
    // UI 업데이트 큐잉
    queueUIUpdate(key) {
        this.eventQueue.push({
            type: 'UI_UPDATE',
            key: key,
            timestamp: Date.now()
        });
        
        if (!this.isProcessing) {
            this.processEventQueue();
        }
    }
    
    // 이벤트 큐 처리
    async processEventQueue() {
        this.isProcessing = true;
        
        while (this.eventQueue.length > 0) {
            const event = this.eventQueue.shift();
            try {
                await this.handleEvent(event);
            } catch (error) {
                console.error('Event processing failed:', error);
            }
        }
        
        this.isProcessing = false;
    }
}
```

// 통계 서비스 (ACIU 교훈 적용)
class StatisticsService {
    static calculateRealTimeStats() {
        try {
            // 1. 사용자 데이터 수집
            const userData = CentralDataManager.getData('userStatistics');
            const sessionData = CentralDataManager.getData('sessionData');
            
            // 2. 실시간 통계 계산
            const stats = {
                totalQuestions: userData?.totalAnswered || 0,
                correctRate: this.calculateCorrectRate(userData),
                averageTime: this.calculateAverageTime(sessionData),
                progress: this.calculateProgress(sessionData),
                achievements: this.calculateAchievements(userData)
            };
            
            // 3. 중앙 아키텍처에 저장
            CentralDataManager.setData('displayStatistics', stats);
            
            return stats;
            
        } catch (error) {
            console.error('Statistics calculation failed:', error);
            throw error;
        }
    }
    
    static calculateCorrectRate(userData) {
        if (!userData?.totalAnswered) return 0;
        return (userData.correctAnswers / userData.totalAnswered * 100).toFixed(1);
    }
    
    static calculateAchievements(userData) {
        const achievements = [];
        
        if (userData?.totalAnswered >= 10) {
            achievements.push({ type: 'first_10', title: '첫 10문제 달성' });
        }
        
        if (userData?.correctAnswers >= 5) {
            achievements.push({ type: 'first_5_correct', title: '첫 5문제 정답' });
        }
        
        return achievements;
    }
}
